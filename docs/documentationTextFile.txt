class dataCollectionAndModification:
    def collectData(ticker, start_date, end_date):
        -collects the data of the ticker from start_date to end_date and returns it
        -ticker is the ticker of the stock whom's data you're collecting (e.g., 'AAPL', 'TSLA', etc.)
        -start_date is that date from which data collection begins in the form of YYYY-MM-DD (e.g., '2026-04-09(April 4th, 2026), '2009-08-17'(August 17th, 2009))
        -end_date is that date which data collection ends on in the form of YYYY-MM-DD (e.g., '2026-04-09(April 4th, 2026), '2009-08-17'(August 17th, 2009))
        -returns the collected data
    def collectAndStoreData(ticker, start_date, end_date, filename):
        -collects the data of the ticker from start_date to end_date and stores it in filename
        -ticker is the ticker of the stock whom's data you're collecting (e.g., 'AAPL', 'TSLA', etc.)
        -start_date is that date from which data collection begins in the form of YYYY-MM-DD (e.g., '2026-04-09(April 4th, 2026), '2009-08-17'(August 17th, 2009))
        -end_date is that date which data collection ends on in the form of YYYY-MM-DD (e.g., '2026-04-09(April 4th, 2026), '2009-08-17'(August 17th, 2009))
        -filename is the name of the file in which the data is stored (e.g., 'stockData.csv', 'dataForResearch.csv') (Recommend CSV for ease of access)
        -returns nothing
    def readData(filename, print):
        -parses the data in filename and prints it if print is true
        -filename is the name of the file in which you want to read (e.g., 'stockData.csv', 'dataForResearch.csv') (Recommend CSV for ease of access)
        -print is a boolean value wher if true it prints the read data
        -returns data parse from filename
    def clearDataFile(filename):
        -clears filename of all data
        -filename is the name of the file which you want cleared
        -returns nothing

    def getEMA(ticker, start_date, end_date, Print=False):
        -Computes the EMA12, EMA26, EMA50, and EMA200, of the ticker, from start_date to end_date, appends them to the array exponentialMovingAverages, and prints the values if Print is true
        -EMA12 stored at position 0 in array, EMA26 stored in position 1, EMA50 stored in position 2, and EMA200 stored in position 3
        -exponentialMovingAverages array cleared everytime function is called
        -ticker is the ticker of the stock whom's data you're collecting (e.g., 'AAPL', 'TSLA', etc.)
        -start_date is that date from which data collection begins in the form of YYYY-MM-DD (e.g., '2026-04-09(April 4th, 2026), '2009-08-17'(August 17th, 2009))
        -end_date is that date which data collection ends on in the form of YYYY-MM-DD (e.g., '2026-04-09(April 4th, 2026), '2009-08-17'(August 17th, 2009))
        -Print is a boolean value which if true leads to the array being printed
        -returns exponentialMovingAverages array
    def getSMA(ticker, start_date, end_date, Print=False):
        -Computes the SMA20, SMA50, SMA100, and SMA200, of the ticker, from start_date to end_date, appends them to the array simpleMovingAverages, and prints the values if Print is true
        -SMA20 stored at position 0 in array, SMA50 stored in position 1, SMA100 stored in position 2, and SMA200 stored in position 3
        -simpleMovingAverages cleared everytime function is called
        -ticker is the ticker of the stock whom's data ou're collecting (e.g., 'AAPL', 'TSLA', etc.)
        -start_date is that date from which data collection begins in the form of YYYY-MM-DD (e.g., '2026-04-09(April 4th, 2026), '2009-08-17'(August 17th, 2009))
        -end_date is that date which data collection ends on in the form of YYYY-MM-DD (e.g., '2026-04-09(April 4th, 2026), '2009-08-17'(August 17th, 2009))

        -Compares the sma values determining if they are a buy, sell, or hold signal

class monteCarloSimulations:
        def callPayoff(ST, K):
            -Determines the payoff of a call option given the stock price at maturity and the strike price
            -ST is the stock price at maturity
            -K is the strike price
            -returns payoff of call option
        def putPayoff(ST, K):
            -Determines the payoff of a put option given the stock price at maturity and the strike price
            -ST is the stock price at maturity
            -K is the strike price
            -returns payoff of put option
        def generateRandomNumbers(n):
            - Generates n standard normally distributed random numbers (Z ~ N(0,1)) using NumPy’s default random number generator
            - n is the number of random values to generate
            - returns a NumPy array of length n containing independent standard normal samples
        def priceOptions(S0, K, r, sigma, T, n_simulations, Print=False):
            -Prices European call and put options using Monte Carlo simulation
            -S0 is the initial stock price
            -K is the strike price
            -r is the risk-free interest rate (annualized)
            -sigma is the volatility of the stock (annualized standard deviation of returns)
            -T is the time to maturity of the option in years
            -n_simulations is the number of Monte Carlo simulations to run
            -if Print equal to true the call and put prices are printed
            -returns callPrice, putPrice

class blackScholesModel:
        def blackScholesCall(S, K, r, sigma, T, Print=False):
            -Prices a European call option using the Black-Scholes formula
            -S is the current stock price
            -K is the strike price
            -r is the risk-free interest rate (annualized)
            -sigma is the volatility of the stock (annualized standard deviation of returns)
            -T is the time to maturity of the option in years
            -if Print equal to true the call price is printed
            -returns callPrice
        def blackScholesPut(S, K, r, sigma, T, Print=False):
            -Prices a European put option using the Black-Scholes formula
            -S is the current stock price
            -K is the strike price
            -r is the risk-free interest rate (annualized)
            -sigma is the volatility of the stock (annualized standard deviation of returns)
            -T is the time to maturity of the option in years
            -if Print equal to true the put price is printed
            -returns putPrice
        def blackScholes(S, K, r, sigma, T, Print=False):
            -Prices both a European call and put option using the Black-Scholes formula and returns both prices as a tuple (callPrice, putPrice)
            -S is the current stock price
            -K is the strike price
            -r is the risk-free interest rate (annualized)
            -sigma is the volatility of the stock (annualized standard deviation of returns)
            -T is the time to maturity of the option in years
            -if Print equal to true both call and put prices are printed
            -returns callPrice, putPrice
